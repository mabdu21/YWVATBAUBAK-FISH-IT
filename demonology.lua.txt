local function Tp(Part, Pos)
    if not Part or not Pos then
        return
    end

    local targetCF

    if typeof(Pos) == "Instance" then

        if Pos:IsA("Model") then
            targetCF = Pos:GetPivot()
        elseif Pos:IsA("BasePart") then
            targetCF = Pos.CFrame
        end

    elseif typeof(Pos) == "Vector3" then
        targetCF = CFrame.new(Pos)

    elseif typeof(Pos) == "CFrame" then
        targetCF = Pos
    end

    if not targetCF then
        warn("Invalid Argument.")
        return
    end

    Part.AssemblyLinearVelocity = Vector3.zero
    Part.AssemblyAngularVelocity = Vector3.zero

    local character = Part:FindFirstAncestorOfClass("Model")
    local humanoid = character and character:FindFirstChildOfClass("Humanoid")

    Part.CFrame = targetCF

    if humanoid then
        humanoid:SetStateEnabled(Enum.HumanoidStateType.FallingDown, false)
        humanoid:SetStateEnabled(Enum.HumanoidStateType.Ragdoll, false)
        humanoid:ChangeState(Enum.HumanoidStateType.GettingUp)
    end
end

local function IsWither()
	for _, p in ipairs(workspace:WaitForChild("Items"):GetChildren()) do
		if p:IsA("Model") and p:GetAttribute("ItemName") == "Flower Pot" then
			local flower = p:FindFirstChild("OptimizedFlower")

			if flower then
				local petals = flower:FindFirstChild("Flower_Petals")

				if petals then
					local c = petals.Color

					if c.R < 0.1 and c.G < 0.1 and c.B < 0.1 then

					end
				end
			end
		end

	return false
end

local function EMFLevel()
	local level = 0

	for _, l in pairs(workspace:WaitForChild("Items"):GetChildren()) do
		if l:IsA("Model")
			and l:GetAttribute("ItemName") == "EMF Reader"
		then
			local indicators = l:FindFirstChild("Indicators")

			if indicators then
				for i = 1, 5 do
					local part = indicators:FindFirstChild(tostring(i))

					if part
						and part:IsA("BasePart")
						and part.Material == Enum.Material.Neon
					then
						level = math.max(level, i)
					end
				end
			end
		end
	end

	return level
end

local function ToggleRoom(room, bringback, force)

    local root = HumanoidRootPart

    if not root or not room then
        return false
    end

    local switch = GetRoomSwitch(room)

    if not switch then
        return false
    end

    local currentState = room:GetAttribute("LightsOn")

    if force ~= nil and currentState == force then
        return true
    end

    local oldCF = root.CFrame

    if bringback then
        Tp(root, switch.PrimaryPart.CFrame * CFrame.new(0,0,-3))
        task.wait(0.15)
    end

    game:GetService("ReplicatedStorage")
        :WaitForChild("Events")
        :WaitForChild("UseLightSwitch")
        :FireServer(room)

    local success = false

    for i = 1, 20 do

        task.wait(0.05)

        local newState = room:GetAttribute("LightsOn")

        if force ~= nil then
            if newState == force then
                success = true
                break
            end
        else
            if newState ~= currentState then
                success = true
                break
            end
        end
    end

    if bringback then
        task.wait(0.1)
        Tp(root, oldCF)
    end

    return success
end

local function ToggleDoor(doorName, bringback, force)

    if bringback then

        task.wait(0.15)
    end

    game:GetService("ReplicatedStorage")
        :WaitForChild("Events")
        :WaitForChild("ClientChangeDoorState")
        :FireServer(door)

    local success = false

    for i = 1, 30 do

        task.wait(0.05)

        local newState = outerDoor:GetAttribute("DoorClosed")

        if force ~= nil then
            if newState == force then
                success = true
                break
            end
        else
            if newState ~= currentState then
                success = true
                break
            end
        end
    end

    if bringback then
        task.wait(0.1)
        Tp(root, oldCF)
    end

    return success
end

local function createBillboard(parent, text)
    local gui = Instance.new("BillboardGui")
    gui.Size = UDim2.new(0, 200, 0, 50)
    gui.AlwaysOnTop = true
    gui.StudsOffset = Vector3.new(0, 2.5, 0)
    gui.Parent = parent

    local label = Instance.new("TextLabel")
    label.Size = UDim2.new(1, 0, 1, 0)
    label.BackgroundTransparency = 1
    label.TextColor3 = Color3.fromRGB(255, 255, 255)
    label.TextStrokeTransparency = 0
    label.TextScaled = false
    label.TextSize = 12
    label.Font = Enum.Font.GothamBold
    label.Text = text
    label.Parent = gui

    return gui, label
end

local function addItemESP(item)
    if not item:IsA("Model") then return end

    local part = item.PrimaryPart or item:FindFirstChildWhichIsA("BasePart")
    if not part then return end

    local name = item:GetAttribute("ItemName") or item.Name

    local gui, label = createBillboard(part, name .. " | 0m")

    ItemESP[item] = {
        gui = gui,
        label = label,
        part = part
    }
end

local function removeItemESP(item)
    local data = ItemESP[item]
    if data then
        if data.gui then data.gui:Destroy() end
    end
    ItemESP[item] = nil
end

                if funct["Players"]["Esp Player"] then
                    task.wait(0.2)
                    addPlayerESP(char)
                end
            end)
        end)

    else

        for player, data in pairs(PlayerESP) do
            if data.gui then
                data.gui:Destroy()
            end
        end
        PlayerESP = {}

        for _, conn in pairs(characterConns) do
            conn:Disconnect()
        end
        characterConns = {}

            playerAddedConn:Disconnect()

        end

        delayStep = 0

        for _, doorModel in ipairs(doors:GetChildren()) do

local flyingObjects = {}
local flyConnection
local addedConnection 

local function clearForces(part)
    for _, v in ipairs(part:GetChildren()) do
        if v:IsA("AlignPosition") or v:IsA("AlignOrientation") or v:IsA("Attachment") then
            v:Destroy()
        end
    end
end

local function isCharacterPart(obj)
    if obj:FindFirstAncestorOfClass("Backpack") then return true end
    local model = obj:FindFirstAncestorOfClass("Model")
    if model and (model.Name == "Ghost" or model:FindFirstChildOfClass("Humanoid")) then
        return true
    end
    return false
end

local function setupHover(obj)
    if not obj:IsA("BasePart") or obj.Anchored then return end

    pcall(function()
        sethiddenproperty(Player, math.huge)
    end)

    local ghostModel = workspace:FindFirstChild("Ghost")
    if ghostModel and obj:IsDescendantOf(ghostModel) then
        return
    end

    if isCharacterPart(obj) or flyingObjects[obj] then return end

    clearForces(obj)

    local att = Instance.new("Attachment")
    att.Name = "HoverAttachment"
    att.Parent = obj

    local alignPos = Instance.new("AlignPosition")
    alignPos.Attachment0 = att
    alignPos.Mode = Enum.PositionAlignmentMode.OneAttachment
    alignPos.MaxForce = 100000
    alignPos.MaxVelocity = math.huge
    alignPos.Responsiveness = 15
    alignPos.Parent = obj

    local alignOri = Instance.new("AlignOrientation")
    alignOri.Attachment0 = att
    alignOri.Mode = Enum.OrientationAlignmentMode.OneAttachment
    alignOri.MaxTorque = 100000
    alignOri.MaxAngularVelocity = math.huge
    alignOri.Responsiveness = 15
    alignOri.Parent = obj

    flyingObjects[obj] = {
        alignPos = alignPos,
        originPos = obj.Position,
        lastPosition = obj.Position
    }
end

Game:CreateToggle({
    Name = "Toggle Hovering Objects",
    CurrentValue = false,
    Flag = "FlyingObj",
    Callback = function(boolean)
        if boolean then
            flyConnection = RunService.Heartbeat:Connect(function()
                pcall(function() sethiddenproperty(Player, "SimulationRadius", math.huge) end)

                local t = os.clock()
                local hoverHeight = 2.5 
                local bobAmplitude = 1
                local bobSpeed = 1.5    

                for obj, data in pairs(flyingObjects) do
                    if obj and obj.Parent and obj:IsA("BasePart") then
                        if isCharacterPart(obj) then
                            clearForces(obj)
                            flyingObjects[obj] = nil
                            continue
                        end

                        local currentPos = obj.Position
                        local targetHoverPos = data.originPos + Vector3.new(0, hoverHeight, 0)
                        
                        if (currentPos - targetHoverPos).Magnitude > 7 then 
                            data.originPos = currentPos
                        end

                        local offset = math.sin(t * bobSpeed) * bobAmplitude
                        data.alignPos.Position = data.originPos + Vector3.new(0, hoverHeight + offset, 0)
                    else
                        flyingObjects[obj] = nil
                    end
                end
            end)

            for _, obj in ipairs(workspace:GetDescendants()) do
                setupHover(obj)
            end

            addedConnection = workspace.DescendantAdded:Connect(function(obj)
                task.wait(0.1) 
                setupHover(obj)
            end)

        else
            if flyConnection then flyConnection:Disconnect() flyConnection = nil end
            if addedConnection then addedConnection:Disconnect() addedConnection = nil end

            for obj, _ in pairs(flyingObjects) do
                if obj and obj.Parent then clearForces(obj) end
            end
            table.clear(flyingObjects)
        end
    end,
})

local GhostItemConnection
local GhostItemData = {}

local function clearGhostItemForces(model)
    for _, v in ipairs(model:GetDescendants()) do
        if v.Name == "GhostItemAttach"
        or v.Name == "GhostItemAlignPos"
        or v.Name == "GhostItemAlignOri" then
            v:Destroy()
        end
    end
end

Items:CreateToggle({
    Name = "Move Selected Items Near Ghost",
    CurrentValue = false,
    Flag = "MoveItemsGhost",

    Callback = function(state)

        if state then

            GhostItemConnection = RunService.Heartbeat:Connect(function()

                pcall(function()
                    sethiddenproperty(Player, "SimulationRadius", math.huge)
                end)

                local ghost = GetGhost()
                if not ghost then
                    return
                end

                local ghostCF = ghost:GetPivot()

                local totalItems = #SelectedGhostItems
                if totalItems <= 0 then
                    return
                end

                local t = os.clock()

                for index, itemName in ipairs(SelectedGhostItems) do

                    local item = FindItem(itemName)

                    if item then

                        local part =
                            item.PrimaryPart
                            or item:FindFirstChildWhichIsA("BasePart")

                        if part then

                            if not GhostItemData[part] then

                                clearGhostItemForces(item)

                                local att = Instance.new("Attachment")
                                att.Name = "GhostItemAttach"
                                att.Parent = part

                                -- POSITION
                                local alignPos = Instance.new("AlignPosition")
                                alignPos.Name = "GhostItemAlignPos"
                                alignPos.Attachment0 = att
                                alignPos.Mode = Enum.PositionAlignmentMode.OneAttachment
                                alignPos.MaxForce = 100000
                                alignPos.Responsiveness = 15
                                alignPos.MaxVelocity = math.huge
                                alignPos.RigidityEnabled = false
                                alignPos.Parent = part

                                -- ROTATION
                                local alignOri = Instance.new("AlignOrientation")
                                alignOri.Name = "GhostItemAlignOri"
                                alignOri.Attachment0 = att
                                alignOri.Mode = Enum.OrientationAlignmentMode.OneAttachment
                                alignOri.MaxTorque = 100000
                                alignOri.Responsiveness = 15
                                alignOri.MaxAngularVelocity = math.huge
                                alignOri.RigidityEnabled = false
                                alignOri.Parent = part

                                GhostItemData[part] = {
                                    AlignPos = alignPos,
                                    AlignOri = alignOri
                                }
                            end

                            local data = GhostItemData[part]

                            -- circle positioning
                            local angle =
                                ((index / totalItems) * math.pi * 2)
                                + (t * 0.75)

                            local radius = 3.5

                            local bobAmplitude = 1
                            local bobSpeed = 2

                            local bobOffset =
                                math.sin((t * bobSpeed) + index)
                                * bobAmplitude

                            local localOffset = Vector3.new(
                                math.cos(angle) * radius,
                                2 + bobOffset,
                                math.sin(angle) * radius
                            )

                            local targetPos =
                                ghostCF:PointToWorldSpace(localOffset)

                            data.AlignPos.Position = targetPos

                            data.AlignOri.CFrame =
                                CFrame.lookAt(
                                    targetPos,
                                    ghostCF.Position
                                )
                        end
                    end
                end
            end)

        else

            if GhostItemConnection then
                GhostItemConnection:Disconnect()
                GhostItemConnection = nil
            end

            for part, _ in pairs(GhostItemData) do
                if part and part.Parent then
                    clearGhostItemForces(part.Parent)
                end
            end

            table.clear(GhostItemData)
        end
    end,
})

                if not funct["Ghost"]["Auto Escape (Chase)"] then
                    return
                end

                local hunting = ghost:GetAttribute("Hunting")

                if hunting and not escaped then
                    escaped = true

                    if HumanoidRootPart then
                        oldCF = HumanoidRootPart.CFrame

                        Tp(
                            HumanoidRootPart,

                                :WaitForChild("Spawns")
                                :WaitForChild("SpawnLocation").CFrame * CFrame.new(0, 5, 0)
                        )
                    end

                elseif not hunting and escaped then
                    escaped = false

                    if oldCF and HumanoidRootPart then
                        Tp(HumanoidRootPart, oldCF)
                    end
                end
            end

            HandleHunt()

            huntConnection =
                ghost:GetAttributeChangedSignal("Hunting"):Connect(HandleHunt)
        end
    end
end)
